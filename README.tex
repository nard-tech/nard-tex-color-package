  \section{color\_commands}

  \verb|color_commands| パッケージを読み込むと、以下のような色に関するコマンドを使うことができる。

  \begin{itemize}
    \item \verb|\ColorName{色名 (TeX の定義)}{色名（表示名）}|
      \begin{quote}
        \textgt{例}
        \begin{itemize}
          \item \verb|\ColorName{red}{赤色}| - \ColorName{red}{赤色}
          \item \verb|\ColorName{green}{緑色}| - \ColorName{green}{緑色}
          \item \verb|\ColorName{blue}{青色}| - \ColorName{blue}{青色}
        \end{itemize}
      \end{quote}
    \item \verb|\RGBValue{R}{G}{B}|
      \begin{quote}
        \textgt{例}
        \begin{itemize}
          \item \verb|\RGBValue{3}{193}{61}| - \RGBValue{3}{193}{61}
        \end{itemize}
      \end{quote}
    \item \verb|\CMYKValue{C}{M}{Y}{K}|
      \begin{quote}
        \textgt{例}
        \begin{itemize}
          \item \verb|\CMYKValue{100}{0}{100}{49}| - \CMYKValue{100}{0}{100}{49}
        \end{itemize}
      \end{quote}
    \item \verb|\HexValue{C}{M}{Y}{K}|
      \begin{quote}
        \textgt{例}
        \begin{itemize}
          \item \verb|\HexValue{ff7e1c}| - \HexValue{ff7e1c}
        \end{itemize}
      \end{quote}
    %
  \end{itemize}
